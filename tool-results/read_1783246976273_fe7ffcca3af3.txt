   459в†’   459в†’        # Use equity (balance + deployed margin) so sizing is consistent

   560в†’   560в†’    def _close_position(self, pos: PaperPosition, reason: str, exit_price: float):
   561в†’   561в†’        """Close a realistic paper position вЂ” includes exit spread, slippage, fee."""
   562в†’   562в†’        if exit_price <= 0:
   563в†’   563в†’            exit_price = pos.current_price or pos.entry_price
   564в†’   564в†’
   565в†’   565в†’        # в”Ђв”Ђв”Ђ Exit price logic в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
   566в†’   566в†’        # SL_HIT and TP_HIT: use exact trigger price (like exchange stop-limit orders).
   567в†’   567в†’        # In live trading, stop-limit orders guarantee the exit price at the trigger level.
   568в†’   568в†’        # Only MAX_HOLD and TRAILING_STOP use simulated fill (market order, no guarantee).
   569в†’   569в†’        if reason in ("SL_HIT", "TP_HIT"):
   570в†’   570в†’            # Calculate the exact trigger price based on PnL% threshold
   571в†’   571в†’            if reason == "SL_HIT":
   572в†’   572в†’                target_pct = -pos.stop_loss_pct
   573в†’   573в†’            else:
   574в†’   574в†’                target_pct = pos.take_profit_pct
   575в†’   575в†’            if pos.side == "LONG":
   576в†’   576в†’                actual_exit_price = pos.entry_price * (1 + target_pct / 100)
   577в†’   577в†’            else:
   578в†’   578в†’                actual_exit_price = pos.entry_price * (1 - target_pct / 100)
   579в†’   579в†’            # Only fee on exit (no spread/slippage вЂ” stop-limit is a maker-like fill)
   580в†’   580в†’            exit_fee = pos.notional_usd * self.executor.taker_fee_pct / 100
   581в†’   581в†’            exit_spread_cost = 0.0
   582в†’   582в†’            exit_slippage_bps = 0.0
   583в†’   583в†’        else:
   584в†’   584в†’            # MAX_HOLD, TRAILING_STOP: market order with spread + slippage
   585в†’   585в†’            exit_result = self.executor.simulate_exit(
   586в†’   586в†’                symbol=pos.symbol,
   587в†’   587в†’                side=pos.side,
   588в†’   588в†’                price=exit_price,
   589в†’   589в†’                notional_usd=pos.notional_usd,
   590в†’   590в†’                is_new_listing=pos.is_new_listing,
   591в†’   591в†’            )
   592в†’   592в†’            actual_exit_price = exit_result.fill_price if exit_result.filled else exit_price
   593в†’   593в†’            exit_fee = exit_result.fee_usd if exit_result.filled else pos.notional_usd * self.executor.taker_fee_pct / 100
   594в†’   594в†’            exit_spread_cost = exit_result.spread_cost_usd if exit_result.filled else 0.0
   595в†’   595в†’            exit_slippage_bps = exit_result.slippage_bps if exit_result.filled else 0.0
   596в†’   596в†’
   597в†’   597в†’        # Recalculate PnL with actual exit price
   598в†’   598в†’        if pos.side == "LONG":
   599в†’   599в†’            gross_pnl_pct = (actual_exit_price - pos.entry_price) / pos.entry_price * 100
   600в†’   600в†’        else:
   601в†’   601в†’            gross_pnl_pct = (pos.entry_price - actual_exit_price) / pos.entry_price * 100
   602в†’   602в†’        actual_pnl = pos.margin_usd * pos.leverage * (gross_pnl_pct / 100)
   603в†’   603в†’
   604в†’   604в†’        # Total fee = entry fee + exit fee (both TAKER)
   605в†’   605в†’        total_fees = pos.entry_fee_usd + exit_fee
   606в†’   606в†’
   607в†’   607в†’        # Net PnL after all fees and spread costs
   608в†’   608в†’        net_pnl = actual_pnl - total_fees
   609в†’   609в†’
   610в†’   610в†’        # Net PnL% = gross PnL% minus fee impact
   611в†’   611в†’        # PnL% should be based on NOTIONAL (position size), not margin.
   612в†’   612в†’        # Example: gross +1.0%, fees 0.10% round-trip в†’ net +0.9%
   613в†’   613в†’        # NOT: net_pnl / margin (that would give 20x leverage-boosted %!)
   614в†’   614в†’        fee_impact_pct = (total_fees / pos.notional_usd * 100) if pos.notional_usd > 0 else 0
   615в†’   615в†’        net_pnl_pct = gross_pnl_pct - fee_impact_pct
   616в†’   616в†’
   617в†’   617в†’        # Update balance
   618в†’   618в†’        self.state.balance_usd += pos.margin_usd + net_pnl
   619в†’   619в†’
   620в†’   620в†’        # Record trade
   621в†’   621в†’        hold_sec = time.time() - pos.invested_at
   622в†’   622в†’        trade = TradeRecord(
   623в†’   623в†’            symbol=pos.symbol,
   624в†’   624в†’            side=pos.side,

   626в†’   626в†’            exit_price=actual_exit_price,

   628в†’   628в†’            margin_usd=pos.margin_usd,
   629в†’   629в†’            notional_usd=pos.notional_usd,
   630в†’   630в†’            pnl_usd=round(net_pnl, 4),
   631в†’   631в†’            pnl_pct=round(net_pnl_pct, 3),
   632в†’   632в†’            hold_sec=round(hold_sec, 1),
   633в†’   633в†’            exit_reason=reason,
   634в†’   634в†’            confidence=pos.confidence,
   635в†’   635в†’            signal_reason=pos.signal_reason,
   636в†’   636в†’            timestamp=time.time(),
   637в†’   637в†’            mid_price_entry=pos.mid_price,
   638в†’   638в†’            spread_cost_usd=round(pos.spread_cost_usd, 4),
   639в†’   639в†’            entry_slippage_bps=round(pos.entry_slippage_bps, 2),
   640в†’   640в†’            exit_slippage_bps=round(exit_slippage_bps, 2),

   642в†’   642в†’            exit_fee_usd=round(exit_fee, 4),
   643в†’   643в†’            total_fees_usd=round(total_fees, 4),
   644в†’   644в†’            fill_qty_ratio=pos.fill_qty_ratio,
   645в†’   645в†’            is_partial_fill=pos.is_partial_fill,
   646в†’   646в†’            max_favorable_pct=round(pos.max_favorable_pct, 3),

   649в†’   649в†’            sl_would_hit=pos.sl_would_hit,
   650в†’   650в†’            price_updates=pos.price_updates,
   651в†’   651в†’        )
   652в†’   652в†’
   653в†’   653в†’        self.state.trade_history.append(trade)
   654в†’   654в†’        self.state.total_pnl += net_pnl
   655в†’   655в†’        self.state.total_fees += total_fees
   656в†’   656в†’        self.state.total_spread_cost += pos.spread_cost_usd
   657в†’   657в†’
   658в†’   658в†’        if net_pnl > 0:
